public class Continhas {
    public static void main(String args[]) {
      int a, b, c;
      a=27;
      b=4;
      c=a+b;
      System.out.println("Valor do c: "+c);
      c=a-b;
      System.out.println("Valor do c: "+c);
      c=a*b;
      System.out.println("Valor do c: "+c);
      c=a/b;
      System.out.println("Valor do c: "+c);
    }
}
